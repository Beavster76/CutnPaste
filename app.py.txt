from fastapi import FastAPI
import os

app = FastAPI(title="CutnPaste API")

@app.get("/")
async def root():
    return {"message": "CutnPaste API is running!"}

@app.get("/health")
async def health():
    return {"status": "healthy"}

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=int(os.environ.get("PORT", 8000)))
